SESSION_POINTAGE(
  id_session_pointage,
  id_pointage,
  debut_session_pointage,
  fin_session_pointage,
  commentaire_session_pointage,
  id_motif_arret_session
)

---------------------------------------------------------------------------

create table session_pointage (
  id_session_pointage bigint generated always as identity primary key,
  id_pointage bigint not null,
  debut_session_pointage timestamp not null default now(),
  fin_session_pointage timestamp,
  commentaire_session_pointage text,
  id_motif_arret_session bigint,

  constraint fk_session_pointage_pointage
    foreign key (id_pointage)
    references pointage(id_pointage),

  constraint session_pointage_dates_check
    check (
      fin_session_pointage is null
      or
      debut_session_pointage < fin_session_pointage
    ),

  constraint fk_session_pointage_motif_arret_session
    foreign key (id_motif_arret_session)
    references motif_arret_session(id_motif_arret_session)
);
